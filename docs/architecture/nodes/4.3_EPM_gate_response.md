import asyncio
import json

EMBEDDING_CHECKPOINT_PATH = "data/rag/embedding_checkpoint.jsonl"

async def embed_with_retry(
    client: openai.AsyncOpenAI,
    texts: list[str],
    max_retries: int = 3,
) -> list[np.ndarray]:
    for attempt in range(max_retries):
        try:
            response = await client.embeddings.create(
                model="text-embedding-3-small", input=texts
            )
            return [normalize_embedding(r.embedding) for r in response.data]
        except Exception:
            if attempt == max_retries - 1:
                raise
            await asyncio.sleep(2 ** attempt)

_clip_model = None
_clip_processor = None

def get_clip_model():
    global _clip_model, _clip_processor
    if _clip_model is None:
        _clip_model = CLIPModel.from_pretrained(CLIP_MODEL_NAME).to("mps")
        _clip_processor = CLIPProcessor.from_pretrained(CLIP_MODEL_NAME)
    return _clip_model, _clip_processor

def release_clip_model():
    global _clip_model, _clip_processor
    del _clip_model, _clip_processor
    _clip_model = None
    _clip_processor = None
    import torch
    torch.mps.empty_cache()

# Usage in morphology comparison endpoint:
async def embed_ecg_figure_for_comparison(image_path: str) -> np.ndarray:
    model, processor = get_clip_model()
    embedding = embed_ecg_figure(image_path, model, processor)
    release_clip_model()  # free MPS memory immediately after use
