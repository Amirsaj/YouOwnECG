### REQUIRED-1: Keyword overlap threshold raised; finding-type citation mismatch logging

**Fix applied to Node 4.5:**

```python
CLAIM_SUPPORT_KEYWORD_THRESHOLD = 0.35  # raised from 0.25

def check_claim_supported(
    claim: str,
    claim_finding_type: str,
    chunks: list[dict],
) -> bool:
    if not chunks:
        return False

    claim_words = set(claim.lower().split())
    stop_words = {"the", "a", "an", "in", "of", "with", "is", "are", "and", "or", "to", "for"}
    claim_words -= stop_words

    for chunk in chunks:
        # Finding-type mismatch soft warning
        chunk_section = chunk.get("section", "").lower()
        if claim_finding_type not in ("general",) and claim_finding_type not in chunk_section:
            log_safety_event(
                event_type="CITATION_SECTION_MISMATCH",
                details=f"Claim finding_type '{claim_finding_type}' not in cited section '{chunk_section}'",
                finding=claim_finding_type,
            )

        chunk_words = set(chunk["text"].lower().split()) - stop_words
        overlap = len(claim_words & chunk_words) / max(len(claim_words), 1)
        if overlap >= CLAIM_SUPPORT_KEYWORD_THRESHOLD:
            return True

    return False
```

For MVP, out-of-scope detection appends a warning without attempting sentence-level truncation:

```python
def apply_out_of_scope_check(text: str, ecg_id: str, finding: str) -> str:
    flagged_patterns = detect_out_of_scope(text)
    if not flagged_patterns:
        return text

    log_safety_event(
        event_type="OUT_OF_SCOPE_DETECTED",
        ecg_id=ecg_id,
        finding=finding,
        details=f"Patterns matched: {flagged_patterns}",
    )
    warning = (
        "\n\n[WARNING: This response may contain content outside ECG interpretation scope. "
        "Clinical review required before acting on any recommendations above.]"
    )
    return text + warning
```

Sentence-level truncation with proper sentence boundary detection is deferred to a post-MVP release with spaCy or NLTK sentence segmentation.
